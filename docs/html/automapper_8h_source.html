141 template <class DestinationType> static QVariant implDefaultMap(const QVariant &source)
142 {
143
144 // list all metatypes in getSiblingFunctions
145 // for (auto it = getSiblingFunctions.begin(); it != getSiblingFunctions.end(); ++it)
146 // {
147 // qDebug() << "from" << it.key().first.name() << "to" << it.key().second.name();
148 // }
149
150 // qDebug() << "Entries of getSiblingFunctions while in implMap: "
151 // << AutoMapper::AutoMapper::getSiblingFunctions.count();
152
153 QVariant destinationVariant = QVariant::fromValue(DestinationType());
154
155 const QMetaObject *sourceMetaObject = source.metaType().metaObject();
156 const QMetaObject *destinationMetaObject = destinationVariant.metaType().metaObject();
157
158 // convert to actual type
159 const void *sourcePointer = source.data();
160 void *destinationPointer = destinationVariant.data();
161
162 int propertyCount = sourceMetaObject->propertyCount();
163
164 for (int i = 0; i < propertyCount; ++i)
165 {
166 QMetaProperty sourceProperty = sourceMetaObject->property(i);
167
168 if (sourceProperty.isReadable())
169 {
170 int destinationPropertyIndex = destinationMetaObject->indexOfProperty(sourceProperty.name());
171
172 if (destinationPropertyIndex >= 0)
173 {
174 const QVariant &value = sourceProperty.readOnGadget(sourcePointer);
175 const QMetaProperty &destinationProperty =
176 destinationMetaObject->property(destinationPropertyIndex);
177
178 if (destinationProperty.isWritable() &&
179 QMetaType::canConvert(value.metaType(), destinationProperty.metaType()))
180 {
181 bool success = destinationProperty.writeOnGadget(destinationPointer, value);
182
183 if (!success)
184 {
185 qWarning() << "Failed to write value" << value << "to destination property"
186 << destinationProperty.name();
187 }
188 }
189
190 else
191 {
192 qCritical("AutoMapper error: Property types do not match. Check the types of the source and "
193 "destination properties.");
194 }
195 }
196 }
197 }
198 return destinationVariant;

201 template <class SourceType, class DestinationType> static QVariant implMap(const SourceType &source)
202 {
203
204 // list all metatypes in getSiblingFunctions
205 // for (auto it = getSiblingFunctions.begin(); it != getSiblingFunctions.end(); ++it)
206 // {
207 // qDebug() << "from" << it.key().first.name() << "to" << it.key().second.name();
208 // }
209 // for (auto it = getDefaultSiblingFunctions.begin(); it != getDefaultSiblingFunctions.end(); ++it)
210 // {
211 // qDebug() << "Default from" << it.key().name() << "to" << it.value()().typeName();
212 // }
213
214 // qDebug() << "Entries of getSiblingFunctions while in implMap: "
215 // << AutoMapper::AutoMapper::getSiblingFunctions.count();
216
217 QVariant destinationVariant = QVariant::fromValue(DestinationType());
218 QVariant sourceVariant = QVariant::fromValue(source);
219
220 const QMetaObject *sourceMetaObject = sourceVariant.metaType().metaObject();
221 const QMetaObject *destinationMetaObject = destinationVariant.metaType().metaObject();
222
223 // convert to actual type
224 const void *sourcePointer = sourceVariant.data();
225 void *destinationPointer = destinationVariant.data();
226
227 int propertyCount = sourceMetaObject->propertyCount();
228
229 for (int i = 0; i < propertyCount; ++i)
230 {
231 QMetaProperty sourceProperty = sourceMetaObject->property(i);
232
233 if (sourceProperty.isReadable())
234 {
235 int destinationPropertyIndex = destinationMetaObject->indexOfProperty(sourceProperty.name());
236
237 if (destinationPropertyIndex >= 0)
238 {
239 const QVariant &value = sourceProperty.readOnGadget(sourcePointer);
240 const QMetaProperty &destinationProperty =
241 destinationMetaObject->property(destinationPropertyIndex);
242
243 QVariant destinationValue;
244
245 if (destinationProperty.isWritable() &&
246 QMetaType::canConvert(value.metaType(), destinationProperty.metaType()))
247 {
248 bool success = destinationProperty.writeOnGadget(destinationPointer, value);
249
250 if (!success)
251 {
252 qWarning() << "Failed to write value" << value << "to destination property"
253 << destinationProperty.name();
254 }
255 }
256
257 else if (destinationProperty.isWritable())
258 {
259 // Check if a conversion function exists for this
260 // property type
261 auto getDefaultSiblingFunction = getDefaultSiblingFunctions.find(value.metaType());
262
263 if (getDefaultSiblingFunction != getDefaultSiblingFunctions.end())
264 {
265 // We have a conversion for this type.
266 if (QString(value.metaType().name()).startsWith("QList<"))
267 {
268 // If it's a QList<QVariant>, process each
269 // QVariant.
270 QList<QVariant> sourceList = value.toList();
271
272 if (sourceList.isEmpty())
273 {
274 // If the list is empty, we can't get the
275 // type of the custom type.
276 // So we can't instantiate a new object of
277 // the custom type.
278 // So we can't call mapImpl recursively.
279 // So we can't convert the list.
280 // So we can't do anything.
281 // So we just return an empty list.
282 continue;
283 }
284
285 // destinationValue = foreignDestinationList;
286
287 auto writeForListIt = writerForListHash.find(
288 MetaTypePair(value.metaType(), getDefaultSiblingFunction.value()().metaType()));
289
290 if (writeForListIt != writerForListHash.end() /*&&
291 source.metaData().getSet(sourceProperty.name())*/)
292 {
293 bool success =
294 writeForListIt.value()(destinationProperty, destinationPointer, sourceList);
295
296 if (!success)
297 {
298 qWarning() << "Failed to write value" << destinationValue
299 << "to destination property" << destinationProperty.name();
300 }
301 }
302 }
303 else
304 {
305 // It's a single QVariant with custom type.
306
307 auto writeIt = writerHash.find(
308 MetaTypePair(value.metaType(), getDefaultSiblingFunction.value()().metaType()));
309
310 if (writeIt != writerHash.end() && source.metaData().getSet(sourceProperty.name()))
311 {
312 bool success = writeIt.value()(destinationProperty, destinationPointer, value);
313
314 if (!success)
315 {
316 qWarning() << "Failed to write value" << destinationValue
317 << "to destination property" << destinationProperty.name();
318 }
319 }
320 }
321 }
322 }
323
324 else
325 {
326 qCritical("AutoMapper error: Property types do not match. Check the types of the source and "
327 "destination properties.");
328 }
329 }
330 }
331 }
332 return destinationVariant;
333 }
